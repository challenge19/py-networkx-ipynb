def spl_1(g):
    spl = []
    for sg in nx.connected_component_subgraphs(g):
        spl.append(nx.average_shortest_path_length(sg))
    return np.mean(spl)


def spl_2(g):
    spl = 0
    N = len(g)
    for sg in nx.connected_component_subgraphs(g):
        n = len(sg)
        if len(sg)>1:
            spl+=nx.average_shortest_path_length(sg) * n*(n-1)
        spl += N*n*(N-n)
    return spl/N/(N-1)

import random

zin = 13

nall = []
for a in ['a','b','c']:
    xx = []
    for i in xrange(n):
        xx.append(a+str(i))
    nodes += xx
    nall.append(xx)

pin = 1.0*zin/(n-1)/2
pout = 1.0*zout/(3*n-1)/2

g = nx.Graph()

for nc in nall:
    for i in nc:
        for j in nc:
            if i==j:
                continue
            p = random.random()
            if p<pin:
                g.add_edge(i,j)
        for j in set(nodes)-set(nc):
            p = random.random()
            if p<pout:
                g.add_edge(i,j)
