def spl_1(g):
    spl = []
    for sg in nx.connected_component_subgraphs(g):
        spl.append(nx.average_shortest_path_length(sg))
    return np.mean(spl)


def spl_2(g):
    spl = 0
    N = len(g)
    for sg in nx.connected_component_subgraphs(g):
        n = len(sg)
        if len(sg)>1:
            spl+=nx.average_shortest_path_length(sg) * n*(n-1)
        spl += N*n*(N-n)
    return spl/N/(N-1)

## Degree Correlation Matrix
def cal_correlation_matrix(g):
    degs = g.degree()
    max_k = max(degs.values())
    emat = np.zeros((max_k+1, max_k+1))
    cmat = np.zeros((max_k+1, max_k+1))
    L = g.number_of_edges()
    for i,j in g.edges():
        ki = degs[i]
        kj = degs[j]
        emat[ki,kj] += 1.0/2/L
        emat[kj,ki] += 1.0/2/L
        cmat[ki,kj] += 2.0*L/ki/kj
        cmat[kj,ki] += 2.0*L/ki/kj
    return emat,cmat

度相关性函数(Degree Correlation Function)¶

$$k_{nn}(k) \equiv \sum_{k'} k'P(k'|k)$$

  • $\mu<0$: Disassortative
  • $\mu=0$: Neutral
  • $\mu>0$: Assortative

匹配性系数 (degree correlation coefficient)¶

degree-degree correlation $$\langle jk \rangle - \langle j \rangle \langle k \rangle=\sum_{jk}jk e_{jk} - \sum_j jq_j \sum_k kq_k=\sum_{jk}jk(e_{jk}-q_jq_k)$$

:$e_{jk}=q_k\delta_{jk}$$q_k$$\sigma^2= \sum_k k^2q^2_k -[\sum_k kq_k]^2$

$$r = \sum_{jk}\frac{jk(e_{jk} - q_jq_k)}{\sigma^2}$$

  • $r<0$: Disassortative
  • $r=0$: Neutral
  • $r>0$: Assortative

## SIR 

def suspect_infect_recovery(G, sources, beta, gamma, T='inf'):
    state = {i:'S' for i in G}
    for s in sources:
        state[s] = 'I'
    if T=='inf':
        dt = 0
    else:
        dt = 1
    iterator = 0
    flag = False
    while iterator < T:
        iterator += dt
        if flag:
            return state
        flag = True
        state_temp = state.copy()
        for i in G:
            if state[i] == 'S':
                nbs = sum([1 for j in G.neighbors(i) if state[j]=="I"])
                if nbs == 0:
                    continue
                if random.random() < (1 - (1-beta)**nbs):
                    state_temp[i] = 'I'
            elif state[i] == 'I':
                flag = False
                if random.random() < gamma:
                    state_temp[i] = 'R'
        state = state_temp.copy()
    return state
